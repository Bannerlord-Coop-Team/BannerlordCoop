[HarmonyPatch(typeof({{ MemberDeclaringType }}))]
public class {{ MemberDeclaringTypeName }}_{{ MemberName }}_PropertySetPrefix : GenericPatches<{{ MemberDeclaringTypeName }}_{{ MemberName }}_PropertySetPrefix, {{ MemberDeclaringType }}>
{
    [HarmonyPatch(nameof({{ MemberDeclaringType }}.{{ MemberName }}), MethodType.Setter)]
    [HarmonyPrefix]
    private static void {{ MemberName }}Prefix(ref {{ MemberDeclaringType }} __instance, ref {{ MemberType }} value)
    {
        if (CallOriginalPolicy.IsOriginalAllowed())
            return;
        
        if (ModInformation.IsClient)
        {
            Logger.Error("Client updated managed {member}", "{{MemberDeclaringType}}.{{MemberName}}");
            return;
        }
        
        if(EqualityComparer<{{MemberType}}>.Default.Equals(__instance.{{MemberName}}, value))
        {
            {{~ if Debug ~}}
            Logger.Debug("[Server] Value same, skipping {member}", "{{MemberDeclaringType}}.{{MemberName}}");
            {{~ end ~}}
            return;

        }

        {{~ if Debug ~}}
        Logger.Debug("[Server] Publishing local set message for {member}", "{{MemberDeclaringType}}.{{MemberName}}");
        {{~ end ~}}

        var message = new {{MemberDeclaringTypeName}}_{{MemberName}}_SetLocalMessage(__instance, value);
        MessageBroker.Instance.Publish(__instance, message);
    }
}
