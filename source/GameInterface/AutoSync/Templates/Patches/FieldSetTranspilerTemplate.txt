[HarmonyTranspiler]
private static IEnumerable<CodeInstruction> {{MemberIdentifier}}_SetTranspiler(IEnumerable<CodeInstruction> instructions)
{
    var fieldIntercept = typeof({{MemberDeclaringTypeName}}_DynamicPatches).GetMethod(nameof({{MemberIdentifier}}_Intercept));

    var fieldInfo = {{MemberIdentifier}}FieldInfo;

    GenericPatchHelpers.FieldInterceptCache.TryAdd(fieldInfo, fieldIntercept);

    foreach (var instruction in instructions)
    {
        if (instruction.StoresField(fieldInfo))
        {
            var callInst = new CodeInstruction(OpCodes.Call, fieldIntercept)
            {
                labels = instruction.labels.ToList(),
                blocks = instruction.blocks.ToList()
            };

            yield return callInst;

        }
        else
        {
            yield return instruction;
        }
    }
}

{
    var currentValue = {{MemberIdentifier}}FieldInfo.GetValue(instance);

    // Skip if value hasn't changed
    if (currentValue != null && currentValue.Equals(newValue)) return;

    if (CallOriginalPolicy.IsOriginalAllowed())
    {
{{~ if DirectAssignment ~}}
        instance.{{MemberName}} = newValue;
{{~ else ~}}
        {{MemberIdentifier}}FieldInfo.SetValue(instance, newValue);
{{~ end ~}}
        return;
    }

    if (ModInformation.IsClient)
    {
        Logger.Error("Client updated managed {type}", "{{MemberName}}");
{{~ if DirectAssignment ~}}
        instance.{{MemberName}} = newValue;
{{~ else ~}}
        {{MemberIdentifier}}FieldInfo.SetValue(instance, newValue);
{{~ end ~}}
        return;
    }

{{~ if Debug ~}}
    Logger.Debug("[Server] Publishing local set message for {type}", "{{MemberName}}");
{{~ end ~}}

    var message = new {{MemberDeclaringTypeName}}_{{MemberIdentifier}}_SetLocalMessage(instance, newValue);
    MessageBroker.Instance.Publish(instance, message);
{{~ if DirectAssignment ~}}
    instance.{{MemberName}} = newValue;
{{~ else ~}}
    {{MemberIdentifier}}FieldInfo.SetValue(instance, newValue);
{{~ end ~}}
}
