namespace DynamicSync;

internal class {{ DeclaringType }}_{{ MemberType }}Handler : IHandler
{
    private readonly IMessageBroker messageBroker;
    private readonly IObjectManager objectManager;
    private readonly INetwork network;
    private readonly ILogger Logger = LogManager.GetLogger<{{ DeclaringType }}_{{ MemberType }}Handler>();

    public ClanLifetimeHandler(IMessageBroker messageBroker, IObjectManager objectManager, INetwork network)
    {
        this.messageBroker = messageBroker;
        this.objectManager = objectManager;
        this.network = network;

        messageBroker.Subscribe<{{ InternalMessage }}>(Handle);
        messageBroker.Subscribe<{{ NetworkMessage }}>(Handle);
    }

    public void Dispose()
    {
        messageBroker.Unsubscribe<{{ InternalMessage }}>(Handle);
        messageBroker.Unsubscribe<{{ NetworkMessage }}>(Handle);
    }

    private void Handle(MessagePayload<{{ InternalMessage }}> payload)
    {
        if (objectManager.TryGetId(payload.What.Instance, out string instanceId) == false)
        {
            Logger.Error("Failed to get {type} id", typeof({{ DeclaringType }}));
            return;
        }

        network.SendAll(new {{ NetworkMessage }}(instanceId, payload.What.Value));
    }

    private void Handle(MessagePayload<{{ NetworkMessage }}> obj)
    {
        var payload = obj.What;

        if (objectManager.TryGetObject<{{ DeclaringType }}>(payload.InstanceId, out var instance) == false)
        {
            Logger.Error("Unable to find {{ DeclaringType }} with string id {stringId}", payload.ValueId);
            return;
        }

        using (new AllowedThread())
        {
            instance.{{ MemberName }} = payload.Value;
        }
    }
}
