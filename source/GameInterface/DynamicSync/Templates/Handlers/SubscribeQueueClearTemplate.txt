messageBroker.Subscribe((MessagePayload<Local_{{ MemberDeclaringType }}_{{ MemberName }}_QueueClear> payload) =>
{
    var data = payload.What;

    if (!objectManager.TryGetIdWithLogging(data.Instance, out var instanceId))
        return;

    {{~ if Debug ~}}
    Logger.Debug("[Server] Broadcasting clear queue {MemberName}", "{{MemberDeclaringType}}.{{MemberName}}");
    {{~ end ~}}

    network.SendAll(new Network_{{ MemberDeclaringType }}_{{ MemberName }}_ClearQueue(instanceId));
});

messageBroker.Subscribe((MessagePayload<Network_{{ MemberDeclaringType}}_{{ MemberName }}_ClearQueue> payload) =>
{
    var data = payload.What;

    if (!objectManager.TryGetObjectWithLogging(data.InstanceId, out {{ MemberType }} instance))
        return;

    try
    {
        using (new AllowedThread())
        {
            {{~ if Debug ~}}
            Logger.Debug("[Client] Clearing Queue {MemberName}", "{{MemberDeclaringType}}.{{MemberName}}");
            {{~ end ~}}
            instance.Clear();
        }
    }
    catch (Exception ex)
    {
        Logger.Error(ex, "Failed to run message handler");
    }
});
